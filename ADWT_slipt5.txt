4. Required Queries
a. List all projects where project_type = 'Web'
db.projects.find({ project_type:"Web" }, {project_name:1,_id:0});

b. List projects with duration > 3 months
db.projects.find({ duration_months:{ $gt:3 } }, {project_name:1,duration_months:1});

c. Count number of employees working on "Inventory System"
db.employees.aggregate([
   { $lookup:{ from:"projects", localField:"project_id", foreignField:"_id", as:"proj" }},
   { $match:{ "proj.project_name":"Inventory System" }},
   { $count:"Total Employees Working" }
]);

d. List the names of projects Mr. Patil is working on
db.employees.aggregate([
   { $match:{ emp_name:"Mr. Patil" }},
   { $lookup:{ from:"projects", localField:"project_id", foreignField:"_id", as:"Project Details" }},
   { $project:{ "Project Details.project_name":1,_id:0 }}
]);
